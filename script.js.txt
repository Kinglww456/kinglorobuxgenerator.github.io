// script.js
let isSubscribed = false; // Kullanıcı abone olmalı
let attemptCount = 0; // Kullanıcının deneme sayısı
let isRestricted = false; // Kullanıcı kısıtlamaya uğradı mı?

// Butonlar
const subscribeBtn = document.getElementById('subscribeBtn');
const claimRobuxBtn = document.getElementById('claimRobuxBtn');
const statusMessage = document.getElementById('statusMessage');
const restrictionMessage = document.getElementById('restrictionMessage');

// Abone olma fonksiyonu
subscribeBtn.addEventListener('click', () => {
    isSubscribed = true;
    subscribeBtn.disabled = true; // Abone olunduğunda abone ol butonu devre dışı kalır
    statusMessage.textContent = "Başarıyla abone oldunuz! Artık Robux alabilirsiniz.";
    claimRobuxBtn.disabled = false; // Robux alma butonu aktif olur
});

// Robux alma fonksiyonu
claimRobuxBtn.addEventListener('click', () => {
    if (!isSubscribed) {
        attemptCount++; // Kullanıcının deneme sayısını arttır

        if (attemptCount === 1) {
            statusMessage.textContent = "Abone olmadan Robux alamazsınız! Lütfen abone olun.";
        } else if (attemptCount === 2) {
            isRestricted = true; // Kısıtlama uygulanacak
            restrictionMessage.style.display = 'block'; // Kısıtlama mesajını göster
            statusMessage.textContent = "İkinci denemede Robux alma işleminiz 1 ay süreyle kısıtlanacaktır.";
            claimRobuxBtn.disabled = true; // Robux alma butonunu devre dışı bırak
        }
    } else if (isSubscribed) {
        statusMessage.textContent = "Robux başarıyla alındı! Haftalık Robux alabilirsiniz.";
    }
});
